# Start containers
docker compose up -d

# Stop containers (data saved)
docker compose down

# Stop and wipe all data
docker compose down -v

# Check running containers
docker ps

# Check logs
docker logs chat-postgres

# Get a terminal inside postgres
docker exec -it chat-postgres psql -U postgres -d chat

-- Inside psql
\dt                  -- list tables
\d users             -- describe table
SELECT * FROM users;
TRUNCATE users CASCADE;
\q                   -- quit


# Migrations
sqlx database create
sqlx database drop
sqlx migrate run
sqlx migrate info
